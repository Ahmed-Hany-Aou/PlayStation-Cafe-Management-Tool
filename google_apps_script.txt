function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    
    // Headers: Timestamp, StationID, StartTime, EndTime, DurationMinutes, GamingTotal, FoodTotal, GrandTotal, FoodItems, CustomerName
    sheet.appendRow([
      new Date(),
      data.stationId || "",
      data.startTime || "",
      data.endTime || "",
      data.durationMinutes || 0,
      data.gamingTotal || 0,
      data.foodTotal || 0,
      data.grandTotal || 0,
      JSON.stringify(data.foodItems) || "[]",
      data.customerName || "Walk-in"
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({"status": "success"}))
      .setMimeType(ContentService.MimeType.JSON);
    
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet() {
  return ContentService.createTextOutput("Service is running. Use POST to send data.");
}
