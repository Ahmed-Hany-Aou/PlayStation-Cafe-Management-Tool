  PlayStation Cafe Management Tool - FINAL v4.0 (SMART FAILSAFE)
  Features: Auto-Header Creation, 12-Column Audit Schema, Deduplication.
*/

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheets()[0];
  try {
    var data = JSON.parse(e.postData.contents);
    var sessionId = data.id.toString();
    
    // 0. AUTO-HEADER: Handle empty sheet safely
    var lastRow = sheet.getLastRow();
    if (lastRow === 0) {
      sheet.appendRow([
        "sessionId", "customerName", "startTime", "duration", "endTime", 
        "stationName", "slotNumber", "gamingTotal", "foodTotal", "grandTotal", 
        "foodCartJson", "auditStartTime"
      ]);
    } else if (sheet.getRange(1, 1).getValue() !== "sessionId") {
      sheet.insertRowBefore(1);
      sheet.getRange(1, 1, 1, 12).setValues([[
        "sessionId", "customerName", "startTime", "duration", "endTime", 
        "stationName", "slotNumber", "gamingTotal", "foodTotal", "grandTotal", 
        "foodCartJson", "auditStartTime"
      ]]);
    }

    // 1. Deduplication (Prevent double saving)
    var lastRow = sheet.getLastRow();
    if (lastRow > 0) {
      var idColumn = sheet.getRange(1, 1, lastRow, 1).getValues();
      for (var i = 0; i < idColumn.length; i++) {
        if (idColumn[i][0].toString() === sessionId) return ContentService.createTextOutput("Duplicate");
      }
    }

    // 2. Map 12 Columns (Strict Audit Schema)
    sheet.appendRow([
      sessionId,                // A
      data.custName || "Guest", // B
      data.startTimeSheet || "-", // C
      data.duration || "-",     // D
      data.endTimeSheet || "-",   // E
      data.stationName || "-",  // F
      data.slotNumber || "-",   // G
      data.gamingTotal || 0,     // H
      data.foodTotal || 0,       // I
      data.grandTotal || 0,      // J
      JSON.stringify(data.foodItems || []), // K
      data.auditStartTime || new Date().toISOString() // L
    ]);

    return ContentService.createTextOutput("Success");
  } catch (err) {
    return ContentService.createTextOutput("Error: " + err.message);
  }
}
