function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheets()[0];
  try {
    var data = JSON.parse(e.postData.contents);
    var sessionId = data.id.toString();
    
    // 1. Deduplication (Ignore if already saved)
    var lastRow = sheet.getLastRow();
    if (lastRow > 0) {
      var idColumn = sheet.getRange(1, 1, lastRow, 1).getValues();
      for (var i = 0; i < idColumn.length; i++) {
        if (idColumn[i][0].toString() === sessionId) {
          return ContentService.createTextOutput(JSON.stringify({status:"Duplicate"}))
                 .setMimeType(ContentService.MimeType.JSON);
        }
      }
    }

    // 2. Map 12 Columns (Strict Schema Alignment)
    // A:sessionId, B:customerName, C:startTime, D:duration, E:endTime, F:stationName, G:slotNumber
    // H:gamingTotal, I:foodTotal, J:grandTotal, K:foodCartJson, L:auditStartTime
    sheet.appendRow([
      sessionId,                          // A
      data.custName || "Guest",           // B
      data.startTimeSheet || "-",         // C
      data.duration || "-",               // D
      data.endTimeSheet || "-",           // E
      data.stationName || "-",            // F
      data.slotNumber || "-",             // G
      data.gamingTotal || 0,              // H
      data.foodTotal || 0,                // I
      data.grandTotal || 0,               // J
      JSON.stringify(data.foodItems || []),// K
      data.auditStartTime || "Audit Fail" // L: The Real immutable start click
    ]);

    return ContentService.createTextOutput(JSON.stringify({status:"Success"}))
           .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({status:"Error", message:err.message}))
           .setMimeType(ContentService.MimeType.JSON);
  }
}

function setup() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheets()[0];
  sheet.clear();
  sheet.appendRow([
    "sessionId", "customerName", "startTime", "duration", "endTime", 
    "stationName", "slotNumber", "gamingTotal", "foodTotal", "grandTotal", "foodCartJson", "auditStartTime"
  ]);
}
